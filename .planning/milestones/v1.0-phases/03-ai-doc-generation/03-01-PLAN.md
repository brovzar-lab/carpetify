    import pLimit from "p-limit";

    // Default concurrency limit of 3 (per D-04: cap at 3 simultaneous Claude API calls)
    export function createConcurrencyPool(concurrency: number = 3) {
      const limit = pLimit(concurrency);
      return {
        // Run a task within the concurrency pool
        run: <T>(fn: () => Promise<T>) => limit(fn),
        // Get current pending count
        get pendingCount() { return limit.pendingCount; },
        get activeCount() { return limit.activeCount; },
      };
    }
