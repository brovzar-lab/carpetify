    ```typescript
    function extractLinks(data: ProjectDataSnapshot): LinkCheckInput[] {
      const links: LinkCheckInput[] = []
      for (const member of data.team) {
        if (!member.filmografia) continue
        for (const entry of member.filmografia) {
          if (entry.enlace && entry.enlace.startsWith('http')) {
            links.push({
              url: entry.enlace,
              label: `${member.nombre_completo} - ${entry.titulo}`,
              verified: false,
              accessible: false,
            })
          }
        }
      }
      return links
    }
    ```

    Key design decisions:
    - Use `filmografia[].enlace` (form-saved data source), NOT `member.enlaces` (top-level array never populated by form, per RESEARCH.md Pitfall 2)
    - Always set `verified: false, accessible: false` because the engine is pure -- no HTTP requests, no React state (per D-12). This means VALD-12 will report "N enlace(s) no verificado(s)" as a warning, which is correct.
    - Guard with `entry.enlace.startsWith('http')` to exclude empty strings or malformed values
    - Label format: `"nombre_completo - titulo"` for human-readable rule output

    Remove the underscore prefix from `_data` parameter name since it is now used.

    ```typescript
    const outputFiles = useMemo((): Array<{ name: string; format: string; sizeMB: number }> => {
      const title = (projectData?.metadata as Record<string, unknown> | undefined)?.titulo_proyecto as string ?? ''
      if (!title || generatedDocs.length === 0) return []

      return generatedDocs
        .filter(d => d.docId in EXPORT_FILE_MAP)
        .map(d => ({
          name: generateFilename(EXPORT_FILE_MAP[d.docId].filenameTemplate, title) + '.pdf',
          format: 'pdf',
          sizeMB: 0, // Size unknown pre-export; filename validation is the critical check
        }))
    }, [generatedDocs, projectData])
    ```
