<objective>
Build the EFICINE score estimation system with two scoring methods: deterministic viability scoring (38 pts, client-side) and AI persona artistic scoring (62 pts, Cloud Function with 5 parallel evaluations).

Purpose: Per D-05/D-06/D-07, the score estimation helps producers understand their competitive position before submission. Viability scores (equipo 2, produccion 12, plan rodaje 10, presupuesto 10, exhibicion 4) are computed from measurable project signals. Artistic scores (guion 40, direccion 12, material visual 10) are estimated by 5 named AI personas who each score independently against the EFICINE rubric.

Output: Pure scoring module + Cloud Function for AI evaluation + 5 persona prompt files + locales for score UI.
</objective>

    export interface ScoreSignal {
      name: string        // What was measured
      present: boolean    // Whether the signal was detected

      maxPoints: number
      estimatedPoints: number
      signals: ScoreSignal[]
      isViability: boolean // true for viability categories, false for artistic
    }

    export interface ImprovementSuggestion {
      points: number       // Estimated point gain
      text: string         // Spanish suggestion text from UI-SPEC
      category: string     // Which scoring category
    }

    export interface PersonaScore {
      personaId: string    // reygadas, marcopolo, pato, leo, alejandro
      personaName: string  // Display name
      scores: Record<string, number> // categoryId -> score
      rationale?: Record<string, string> // categoryId -> brief reasoning
    }

    export interface ScoreEstimate {
      viability: ScoreCategory[]      // 5 deterministic categories
      artistic: ScoreCategory[]       // 3 AI-estimated categories
      personaScores: PersonaScore[]   // 5 individual persona results
      bonusPoints: number             // 0 or 5
      bonusCategory: string | null    // Which category earned bonus
      totalViability: number          // Sum of viability scores
      totalArtistic: number           // Sum of artistic scores (averaged across personas)
      totalEstimated: number          // viability + artistic + bonus
      maxPossible: number             // 100
      passingThreshold: number        // 90
      meetsThreshold: boolean
      improvements: ImprovementSuggestion[]
    }
