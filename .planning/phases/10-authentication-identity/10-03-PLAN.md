**Step 1: Create requireAuth helper at functions/src/auth/requireAuth.ts**

Create `functions/src/auth/` directory.

```typescript
import { HttpsError, type CallableRequest } from 'firebase-functions/v2/https';
import { getFirestore } from 'firebase-admin/firestore';

/**
 * Validates that the caller is authenticated.
 * Throws HttpsError('unauthenticated') if no valid auth token.
 * Returns the caller's UID.
 */
export function requireAuth(request: CallableRequest<unknown>): string {
  const uid = request.auth?.uid;
  if (!uid) {
    throw new HttpsError('unauthenticated', 'Autenticacion requerida.');
  }
  return uid;
}

/**
 * Validates that the caller owns the specified project.
 * Must be called AFTER requireAuth.
 * Throws HttpsError('permission-denied') if not the owner.
 * Returns the project data for downstream use.
 */
export async function requireProjectAccess(
  uid: string,
  projectId: string,
): Promise<FirebaseFirestore.DocumentData> {
  const db = getFirestore();
  const projectDoc = await db.doc(`projects/${projectId}`).get();

  if (!projectDoc.exists) {
    throw new HttpsError('not-found', 'Proyecto no encontrado.');
  }

  const data = projectDoc.data()!;

  // Owner check (per D-15). In Phase 10, only owner has access.
  // Phase 11 will extend this to check collaborators array.
  if (data.ownerId !== uid) {
    throw new HttpsError('permission-denied', 'No tienes acceso a este proyecto.');
  }

  return data;
}
```

Note on `CallableRequest` type: in Firebase Functions v2, the `onCall` handler receives a `CallableRequest<T>`. The `.auth` property is `{ uid: string; token: DecodedIdToken } | undefined`. The type import from `firebase-functions/v2/https` should work. If the exact generic is problematic, use `CallableRequest<unknown>`.

**Step 1: Rewrite firestore.rules**

Per D-16: deploy AFTER data migration (this is Wave 3, after Plan 02 which runs migration).
Per D-17: authenticated users can only read/write projects where `ownerId == request.auth.uid`.
Per Pitfall #10: use MINIMAL document reads in rules. For Phase 10, we only check `ownerId` on the project doc itself (zero extra reads for subcollections since we use the parent doc's data).

CRITICAL: In Phase 10, we only have owner-based access (no collaborators yet -- that's Phase 11). Keep rules simple.

```
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Helper: is the user authenticated?
    function isAuth() {
      return request.auth != null;
    }

    // Helper: is user the project owner?
    // Uses get() to read the project document -- costs 1 read per rule evaluation
    function isProjectOwner(projectId) {
      return isAuth() &&
        get(/databases/$(database)/documents/projects/$(projectId)).data.ownerId == request.auth.uid;
    }

    // Projects: only owner can read/write
    match /projects/{projectId} {
      // Read: owner can read their own project doc
      // The ownerId check uses resource.data (free -- no extra read)
      allow read: if isAuth() && resource.data.ownerId == request.auth.uid;
      // Create: any authenticated user can create (they set themselves as owner)
      allow create: if isAuth() && request.resource.data.ownerId == request.auth.uid;
      // Update: only owner
      allow update: if isAuth() && resource.data.ownerId == request.auth.uid;
      // Delete: only owner
      allow delete: if isAuth() && resource.data.ownerId == request.auth.uid;

      // All subcollections: inherit project-level owner check
      // Uses get() for 1 read to check parent project's ownerId
      match /{subcollection}/{docId} {
        allow read, write: if isProjectOwner(projectId);
      }

      // Nested subcollections (e.g., generated/{docId}/history/{versionId})
      match /{subcollection}/{docId}/{nestedSub}/{nestedDocId} {
        allow read, write: if isProjectOwner(projectId);
      }
    }

    // Organizations: members can read/write
    match /organizations/{orgId} {
      allow read: if isAuth() &&
        request.auth.uid in resource.data.members;
      allow create: if isAuth();
      allow update: if isAuth() &&
        request.auth.uid in resource.data.members;
      allow delete: if false; // Organizations cannot be deleted via client

      // ERPI settings under organization
      match /erpi_settings/{settingsId} {
        allow read, write: if isAuth() &&
          request.auth.uid in get(/databases/$(database)/documents/organizations/$(orgId)).data.members;
      }
    }

    // User profiles: users can read any profile, write only their own
    match /users/{userId} {
      allow read: if isAuth();
      allow create: if isAuth() && request.auth.uid == userId;
      allow update: if isAuth() && request.auth.uid == userId;
      allow delete: if false;
    }

    // Legacy ERPI settings path (read-only fallback for migration)
    match /erpi_settings/{settingsId} {
      allow read: if isAuth();
      allow write: if false; // No writes to legacy path
    }
  }
}
```

Key design decisions:
- Project document uses `resource.data.ownerId` for reads (FREE -- no extra get())
- Subcollections use `isProjectOwner()` which does 1 `get()` call (within 10-read limit)
- Organizations checked via `members` array on the org document itself (FREE for the org doc, 1 get() for ERPI sub)
- Legacy `erpi_settings/default` path remains readable (for any admin migration verification) but not writable
- No `collaborators` check yet -- that's Phase 11 (D-17 says ownerId only for Phase 10)

**Step 2: Rewrite storage.rules**

Per D-18: authenticated users can only access files under `projects/{projectId}/` where they own the project.

```
rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Default: deny all
    match /{allPaths=**} {
      allow read, write: if false;
    }

    // Project files: only project owner can read/write
    // Storage rules cannot do Firestore get() calls, so we rely on:
    // 1. Cloud Functions validation (files are uploaded via Cloud Functions)
    // 2. Authenticated access as minimum gate
    // Phase 11 will add custom claims for more granular access
    match /projects/{projectId}/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }
  }
}
```

Note: Firebase Storage rules cannot call `firestore.get()` to check project ownership. The ownership enforcement for Storage relies on:
1. Cloud Functions that handle uploads (extractScreenplay) already verify ownership
2. The frontend only constructs paths for projects the user owns
3. For Phase 10 (single-owner, no collaborators), `request.auth != null` is sufficient since the dashboard only shows owned projects

This is documented as a known limitation. Phase 11 can add custom claims to storage rules for tighter access.
