**buildChangeSummary(screen, changedFields)** -- returns Spanish summary string:
- 1 field: `"Actualizo ${FIELD_LABELS[field] || field} en ${SCREEN_LABELS[screen] || screen}"`
- N fields: `"Actualizo ${changedFields.length} campos en ${SCREEN_LABELS[screen] || screen}"`

**writeActivityEntry(projectId, entry)** -- writes a single ActivityLogEntry to `projects/{projectId}/activity_log` subcollection using `addDoc` with `serverTimestamp()` for `createdAt`. Entry does NOT include `id` or `createdAt` (Firestore generates those).

**coalesceOrCreate(projectId, newEntry, user)** -- implements Pitfall 1 coalescing:
1. Query `projects/${projectId}/activity_log` for entries where `userId == user.uid` AND `screen == newEntry.screen` ordered by `createdAt desc`, limit 1.
2. If a matching entry exists AND its `createdAt` is less than 30 seconds ago (compare with `new Date()`), update that document: merge `changedFields` arrays (union -- no duplicates via `Array.from(new Set([...existing, ...new]))`), update `summary` with `buildChangeSummary`, update `createdAt` to `serverTimestamp()`.
3. If no match or too old, call `writeActivityEntry` with a new entry.
4. This function is fire-and-forget -- catch errors and log to console, never throw. Activity logging must not block the user's save.

**2. Create `src/hooks/useActivityLog.ts`** -- real-time activity log subscription with pagination:

```typescript
import { useEffect, useState, useCallback, useRef } from 'react'
import {
  collection, query, orderBy, limit, onSnapshot, startAfter,
  getDocs, type QueryDocumentSnapshot, type DocumentData, Timestamp,
} from 'firebase/firestore'
import { db } from '@/lib/firebase'
import type { ActivityLogEntry } from '@/services/activityLog'

export function useActivityLog(projectId: string, pageSize = 50) {
  const [entries, setEntries] = useState<ActivityLogEntry[]>([])
  const [loading, setLoading] = useState(true)
  const [hasMore, setHasMore] = useState(true)
  const [loadingMore, setLoadingMore] = useState(false)
  const lastDocRef = useRef<QueryDocumentSnapshot<DocumentData> | null>(null)

  // Real-time subscription to the latest entries
  useEffect(() => {
    if (!projectId) return

    const q = query(
      collection(db, `projects/${projectId}/activity_log`),
      orderBy('createdAt', 'desc'),
      limit(pageSize),
    )

    const unsubscribe = onSnapshot(q, (snap) => {
      const items = snap.docs.map((doc) => {
        const data = doc.data()
        return {
          id: doc.id,
          userId: data.userId,
          displayName: data.displayName,
          photoURL: data.photoURL ?? null,
          userRole: data.userRole ?? '',
          screen: data.screen,
          action: data.action,
          changedFields: data.changedFields ?? [],
          summary: data.summary ?? '',
          createdAt: data.createdAt?.toDate?.() ?? new Date(),
          metadata: data.metadata,
        } as ActivityLogEntry
      })
      setEntries(items)
      setLoading(false)
      setHasMore(snap.docs.length >= pageSize)
      if (snap.docs.length > 0) {
        lastDocRef.current = snap.docs[snap.docs.length - 1]
      }
    })

    return unsubscribe
  }, [projectId, pageSize])

  // Load more (pagination via startAfter cursor)
  const loadMore = useCallback(async () => {
    if (!lastDocRef.current || !projectId || loadingMore) return

    setLoadingMore(true)
    const q = query(
      collection(db, `projects/${projectId}/activity_log`),
      orderBy('createdAt', 'desc'),
      startAfter(lastDocRef.current),
      limit(pageSize),
    )

    const snap = await getDocs(q)
    const moreItems = snap.docs.map((doc) => {
      const data = doc.data()
      return {
        id: doc.id,
        userId: data.userId,
        displayName: data.displayName,
        photoURL: data.photoURL ?? null,
        userRole: data.userRole ?? '',
        screen: data.screen,
        action: data.action,
        changedFields: data.changedFields ?? [],
        summary: data.summary ?? '',
        createdAt: data.createdAt?.toDate?.() ?? new Date(),
        metadata: data.metadata,
      } as ActivityLogEntry
    })

    setEntries((prev) => [...prev, ...moreItems])
    setHasMore(snap.docs.length >= pageSize)
    if (snap.docs.length > 0) {
      lastDocRef.current = snap.docs[snap.docs.length - 1]
    }
    setLoadingMore(false)
  }, [projectId, pageSize, loadingMore])

  return { entries, loading, hasMore, loadMore, loadingMore }
}
```

The hook returns `{ entries, loading, hasMore, loadMore, loadingMore }` for the ActivityTab to consume.
