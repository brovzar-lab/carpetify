    ```typescript
    import type { ProjectDataSnapshot, ValidationReport, ValidationResult } from './types'
    // Import all 14 rule functions...

    /** Rule IDs that fire on every data change (instant tier per D-11) */
    export const INSTANT_RULE_IDS = [
      'VALD-01', 'VALD-02', 'VALD-03', 'VALD-04', 'VALD-05',
      'VALD-06', 'VALD-07', 'VALD-08', 'VALD-09',
      'VALD-12', 'VALD-13', 'VALD-17',
    ] as const

    /** Rule IDs that fire only on document generation/edit (medium tier per D-11) */
    export const MEDIUM_RULE_IDS = ['VALD-10', 'VALD-11'] as const

    /**
     * Run only the 12 instant-tier rules.
     * Called by useValidation on every data change (300ms debounce).
     */
    export function runInstantRules(data: ProjectDataSnapshot): ValidationReport {
      const results: ValidationResult[] = [
        validateFinancialReconciliation(data.budgetTotalCentavos, data.cashFlowTotalCentavos, data.esquemaTotalCentavos),
        validateTitleConsistency(data.metadata.titulo_proyecto, data.generatedDocs, data.uploadedDocs),
        validateFeeCrossMatch(data.feesFromContracts, data.feesFromBudget, data.feesFromCashFlow),
        validateDateCompliance(data.uploadedDocs, data.metadata.periodo_registro),
        validateEficineCompliance(data.metadata.costo_total_proyecto_centavos, data.financials),
        validateDocumentCompleteness(data.generatedDocs, data.uploadedDocs, data.metadata),
        validateExperienceThresholds(data.team, data.metadata.categoria_cinematografica),
        validateErpiEligibility(data.erpiSettings, data.submissionsThisPeriod, data.projectAttempts),
        validateFileFormatCompliance(data.outputFiles),
        validateHyperlinkAccessibility(/* links from snapshot */),
        validateBonusEligibility(/* bonus input from snapshot */),
        validateDocumentExpiration(data.uploadedDocs, data.metadata.periodo_registro),
      ]
      return buildReport(results)
    }

    /**
     * Run only the 2 medium-tier rules (VALD-10, VALD-11).
     * Called by useValidation only when generatedDocs timestamps change.
     */
    export function runMediumRules(data: ProjectDataSnapshot): ValidationReport {
      const results: ValidationResult[] = [
        validateProhibitedExpenditure(data.cashFlowLineItems),
        validateRutaCriticaSync(/* ruta critica + cash flow stages from snapshot */),
      ]
      return buildReport(results)
    }

    /**
     * Run all 14 rules. Convenience for initial load and export gating.
     */
    export function runAllRules(data: ProjectDataSnapshot): ValidationReport {
      const instant = runInstantRules(data)
      const medium = runMediumRules(data)
      return buildReport([...instant.results, ...medium.results])
    }

    function buildReport(results: ValidationResult[]): ValidationReport {
      return {
        results,
        blockers: results.filter(r => r.severity === 'blocker' && r.status === 'fail'),
        warnings: results.filter(r => r.severity === 'warning' && r.status === 'fail'),
        passed: results.filter(r => r.status === 'pass'),
        skipped: results.filter(r => r.status === 'skip'),
        canExport: results.filter(r => r.severity === 'blocker' && r.status === 'fail').length === 0,
        timestamp: new Date(),
      }
    }
    ```
