    ```typescript
    import { getFunctions, httpsCallable } from 'firebase/functions'

    const estimateScoreFn = httpsCallable(getFunctions(), 'estimateScore', {
      timeout: 120_000, // 120s client-side timeout
    })

    const handleEvaluate = async () => {
      setEvaluating(true)
      setError(null)
      try {
        const result = await estimateScoreFn({ projectId })
        setPersonaScores(result.data.personaScores)
      } catch (err) {
        setError(es.scoring.evaluationError)
        // Show retry button: "Reintentar evaluacion"
      } finally {
        setEvaluating(false)
      }
    }
    ```
